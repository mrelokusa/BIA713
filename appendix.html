# Helper function: mode
get_mode <- function(x) {
  ux <- unique(x)
  ux[which.max(tabulate(match(x, ux)))]
}

# Function to build frequency table using Sturges rule
make_freq_table <- function(data) {
  x      <- data$HEADINC
  k      <- ceiling(1 + 3.322 * log10(length(x)))
  breaks <- pretty(x, n = k)
  cuts   <- cut(x, breaks = breaks, include.lowest = TRUE, right = FALSE)
  f      <- table(cuts)
  f_pct  <- round(prop.table(f) * 100, 2)
  F_cum  <- cumsum(f)
  F_pct_cum <- round(cumsum(prop.table(f)) * 100, 2)
  data.frame(Interval = names(f), f = as.integer(f),
             f_pct = as.numeric(f_pct), F = as.integer(F_cum),
             F_pct = as.numeric(F_pct_cum))
}

wc_s <- hh_sample[hh_sample$PROVINCE == "WC", ]
ec_s <- hh_sample[hh_sample$PROVINCE == "EC", ]

cat("Western Cape\n"); print(make_freq_table(wc_s))
